================================================================================
USAGE EXAMPLE
================================================================================

import WalletAuthModal from "@/components/WalletAuthModal";
import UserProfileDropdown from "@/components/UserProfileDropdown";

export default function App() {
  const [user, setUser] = useState<{
    address: string;
    blockchain: "solana" | "ethereum";
  } | null>(null);

  const [isAuthOpen, setIsAuthOpen] = useState(false);

  return (
    <div>
      {user ? (
        <UserProfileDropdown
          address={user.address}
          blockchain={user.blockchain}
          onSignOut={() => setUser(null)}
        />
      ) : (
        <button onClick={() => setIsAuthOpen(true)}>
          Sign In with Wallet
        </button>
      )}

      <WalletAuthModal
        isOpen={isAuthOpen}
        onClose={() => setIsAuthOpen(false)}
        onAuthenticated={(address, blockchain) => {
          setUser({ address, blockchain });
          setIsAuthOpen(false);
        }}
      />
    </div>
  );
}

================================================================================
AUTHENTICATION FLOW
================================================================================

SOLANA FLOW:
1. User selects "Solana" tab
2. Selects wallet (Phantom/Solflare/Coinbase)
3. Wallet connects via useWallet() hook
4. System generates auth message with nonce
5. User signs message with signMessage()
6. Signature is verified
7. User authenticated
8. Profile displayed

ETHEREUM FLOW:
1. User selects "Ethereum" tab
2. Selects wallet (MetaMask/WalletConnect/Coinbase)
3. System calls eth_requestAccounts
4. Wallet connects and returns address
5. System generates auth message with nonce
6. User signs message with personal_sign
7. Signature is verified
8. User authenticated
9. Profile displayed

ERROR FLOW:
1. Connection fails or user rejects
2. Error state shown with details
3. User can click "Try Again"
4. Returns to wallet selection

================================================================================
KEY UTILITIES
================================================================================

truncateAddress(address, chars?)
  └─ Truncates wallet address to readable format
  └─ Example: "0x1234...5678"

generateIdenticon(address)
  └─ Generates deterministic SVG avatar
  └─ Returns SVG data URI
  └─ Unique per address

signAuthMessage(address)
  └─ Generates message for signing
  └─ Includes nonce and timestamp
  └─ Safe for replay prevention

verifySignature(address, message, signature)
  └─ Verifies signed message
  └─ Mock implementation in library
  └─ Production version uses ethers.js

isValidSolanaAddress(address)
  └─ Validates Solana address format
  └─ Checks base58 encoding
  └─ Verifies length requirements

isValidEthereumAddress(address)
  └─ Validates Ethereum address format
  └─ Checks 0x prefix and hex encoding
  └─ Verifies checksum

getNetworkBadgeColor(network)
  └─ Returns Tailwind CSS classes
  └─ Color-codes by network (Mainnet/Testnet)
  └─ Used in profile dropdown

formatNetwork(network)
  └─ Formats network name for display
  └─ Maps network IDs to readable names
